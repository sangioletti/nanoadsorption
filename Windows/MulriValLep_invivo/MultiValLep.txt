# Define the design of the multivalent nanoparticle
#NmonoLong = Number of monomers in ligand conjugated PEG chain
#NmonoPEG2k = Number of monomers in PEG2000 chain
amono = amono * nm # Monomer size in PEG chains
PEG2k = PEG2k * nm # Unperturbed end-to-end distance PEG2000
PEG_max_extension = PEG_max_extension * nm # Max extension of the ligand conjugated PEG chain (dense brush)
KD = KD * nM # Dissociation constant in solution between ligand-receptor
K_bind_0 = KD**(-1) # Binding constant in solution between ligand-receptor
R_NP = d_NP / 2 * nm # Nanoparticle radius
v_bind = np.pi / 3.0 * ((( R_NP + PEG_max_extension) ** 2 - ( R_NP + PEG2k) ** 2 ) * ( R_NP + PEG2k )- 2.0 * R_NP ** 3 * ( 1.0 - (R_NP + PEG2k) / (R_NP + PEG_max_extension)))
sigma_P2K = 1.0 / ( APEG2k * nm2 ) # Surface density of PEG2000 chains
sigma_R = sigma_R / um2 # Receptor density

#****Dosing particles intravenously to animal******
Npdosing = Npdosing / mL # Number of particles in dosing solution per mL [mL-1]
Vdosing = Vdosing * mL # Dosing volume for each animal [mL] (e.g. 5 mL/kg and animal weight 0.02 kg)
#fROI = Fraction of dosed particles that ends up in the tissue ROI
VROI = VROI * mL # Volume of tissue ROI in animal [mL]
NP_conc = Npdosing * Vdosing * fROI / VROI # Number of particles per mL in ROI of animal tissue [mL-1]
#N_cells = Number of cells in the tissue ROI
cell_conc = N_cells / VROI # Cell concentration in tissue
A_cell = np.pi * d_cell ** 2 * um2 # Cell area

#print(f"N_ligands_min: {N_ligands_min}, N_ligands_max: {N_ligands_max}")
sigma_L_min = N_ligands_min / ( 4.0 * np.pi * R_NP**2 )
sigma_L_max = N_ligands_max / ( 4.0 * np.pi * R_NP**2 )
#print(f"sigma_L_min: {sigma_L_min}, sigma_L_max: {sigma_L_max}")

def sigma_L_to_N_ligands(sigma_L):
    return sigma_L * ( 4.0 * np.pi * R_NP**2 )

# Control output
verbose = False
n_sampling_points = npoints #Number of sampling points for the receptor surface density

def process_sigma_L(sigma_L, K_bind_0, nonspec_interaction, binder_linear_size, verbose):
    start_time = time.time()

    data_polymers = {
            "ligands": {
                "N": NmonoLong,
                "a": amono,
                "name": "ligands",
                "sigma": sigma_L
            },
            "short": {
                "N": NmonoPEG2k,
                "a": amono,
                "name": "PEG2k",
                "sigma": sigma_P2K
            }
        }

    # Create the system
    system = MultivalentBinding(
        kT=kT,
        R_NP = R_NP,
        data_polymers = data_polymers,
        polymer_model = "Flory-exact",
        binding_model = "exact",
        A_cell = A_cell,
        NP_conc = NP_conc,
        cell_conc = cell_conc,
        nonspec_interaction = nonspec_interaction,
        binder_linear_size = binder_linear_size,
    )

    # Now we can calculate the adsorbed fraction
    adsorbed_fraction = system.calculate_bound_fraction(
        K_bind_0=K_bind_0,
        sigma_R=sigma_R,
        fluctuations=True,
        depletion=True,
        max_factor=4,
        verbose=verbose
    )
    # Print the adsorbed fraction
    out1 = float(sigma_L_to_N_ligands(sigma_L))
    out2 = float(adsorbed_fraction)
    out3 = float(adsorbed_fraction) * (Npdosing * Vdosing * fROI / N_cells)
    # make sure that the system is deleted and reinstantiated for the next iteration
    del system
    end_time = time.time()
    return out1, out2, out3, end_time - start_time
